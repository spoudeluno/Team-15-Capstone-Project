/**
 * Represents the configuration settings for a proposal, including columns to show, template ID, country code, and signature visibility.
 */
export default class ProposalConfiguration {
    /**
     * Array of column names to display in the proposal.
     * @type {string[]}
     */
    columnsToShow = ['id', 'builderId', 'itemTitle', 'unitCost'];

    /**
     * The ID of the template to use for the proposal.
     * @type {number}
     */
    templateID = 1;

    /**
     * The country code for the proposal.
     * @type {string}
     */
    countryCode = 'US';

    /**
     * Determines whether the signature should be displayed in the proposal.
     * @type {boolean}
     */
    showSignature = true;
    /**
     * Determine which contact information to show in the proposal.
     * @type {object}
     */
    contactConfiguration = {
        showJobAddress: true,
        jobAddressToggles: [
            true,
            true,
            true,
            true
        ],
        showContactAddress: true,
        contactAddressToggles: [
            true,
            true,
            true,
            true
        ],
        showBuilderAddress: true,
        builderAddressToggles: [
            true,
            true,
            true,
            true
        ],
        showBuilderContact: true,
        builderContactToggles: [
            true,

        ],
        showOwnerContact: true,
        ownerContactToggles: [
            true,

        ]
    }
}

/**
 * Represents the HTML content and visibility settings for a proposal.
 */
class ProposalHTML {
    /**
     * The HTML content of the proposal.
     * @type {string}
     */
    html = '';
    /**
     * Determine if there is a preset proposal templete.
     * @type {boolean}
     */
    starterTemplate = true;
    /**
     * The HTML content of the proposal when the customization imformation are provided.
     * @type {string}
     */
    generatedProp = ''

    /**
     * Determines whether additional properties are shown.
     * @type {boolean}
     */
    showProps = false;

    /**
     * Determines which columns are showing.
     * @type {string[]}
     */
    columnElements = [];

    /**
     * Determines the width for the columns.
     * @type {int[]}
     */
    columnWidths = [];

    touched = false;
    /**
     * Determines if any data has been sent form the export page to the template page.
     * @type {boolean}
     */
    fetchBackCSS = false;
}

/**
 * The proposal HTML content.
 * @type {ProposalHTML}
 */
export var proposal = new ProposalHTML();

/**
 * The proposal configuration settings.
 * @type {ProposalConfiguration}
 */
export var proposalConfiguration = new ProposalConfiguration();
