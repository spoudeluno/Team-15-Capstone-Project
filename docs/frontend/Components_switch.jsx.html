/**
 * A custom switch component.
 * @param {Object} data - The data object containing switch configuration.
 * @param {string} data.id - The id of the switch input element.
 * @param {string} data.label - The label text for the switch.
 * @param {Function} data.onChange - The function to be called when the switch state changes.
 * @returns {JSX.Element} A JSX element representing the switch component.
 */
export function Switch(data) {

    /**
     * Toggles the state of the switch input element.
     */
    function toggle() {
        const input = document.getElementById(data.id);
        if (input.checked) {
            input.checked = false;
        } else {
            input.checked = true;
        }
        data.onChange();
    }

    return (
            <div className="switchItemCont pointer noHighlight" onClick={() => toggle()}>
                <label>{data.label}</label>
                <input type="checkbox" className="switch unclickable" id={data.id} onChange={data.onChange} name={data.name}></input>
            </div>
    );
}
