import { CountrySelect } from '../countrySelect';
import { useEffect } from 'react';
import { Switch } from '../switch';
import { proposalConfiguration } from '../../DataModels/ProposalConfiguration.js';
import { initializeCheckboxes } from '../../Functions/multiSelectHelper';

/**
 * A React component that provides the components that can be customize.
 *
 * @param {Object} data - The data object containing information about the customizatioin.
 * @returns {JSX.Element} The rendered CountrySelect component. 
 */
export default function GeneralCustomization(data) {
    useEffect(() => {
        initializeCheckboxes();
    }, []);


    const contactConfigOptions = [
        { id: 'showJobAddress', label: 'Show Job Address' },
        { id: 'showContactAdress', label: 'Show Contact Address' },
        { id: 'showOwnerContact', label: 'Show Owner Contact Info' },
        { id: 'showBuilderAddress', label: 'Show Builder Address' },
        { id: 'showBuilderContact', label: 'Show Builder Contact Info' }
    ];

    function getContactChecked(id) {
        switch (id) {
            case 'showJobAddress':
                return proposalConfiguration.contactConfiguration.showJobAddress;
            case 'showContactAdress':
                return proposalConfiguration.contactConfiguration.showContactAdress;
            case 'showOwnerContact':
                return proposalConfiguration.contactConfiguration.showOwnerContact;
            case 'showBuilderAddress':
                return proposalConfiguration.contactConfiguration.showBuilderAddress;
            case 'showBuilderContact':
                return proposalConfiguration.contactConfiguration.showBuilderContact;
            default:
                return true;
        }
    }

    return (
        <div className="tabContent generalCust">
            <h3 id='desc'>Please select the fields to display</h3>
            <CountrySelect onChange={data.onChange} />
            <div className="switchCont">
                <Switch onChange={data.onChange} label="Show signature fields" id="showSignature" name="showSignature" checked={proposalConfiguration.showSignature} />
                {contactConfigOptions.map(option => (
                    <Switch key={option.id} id={option.id} onChange={data.onChange} label={option.label} name="contactConfig" checked={getContactChecked(option.id)} />
                ))}
            </div>
        </div>
    );
}
