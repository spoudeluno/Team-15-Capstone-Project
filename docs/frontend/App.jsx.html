import { Routes, Route } from "react-router-dom";
import StartPage from "./Components/startPage";
import Templates from "./Components/templates";
import Tiptap from "./Components/export";
import './App.css';

/**
 * The main component of the application responsible for routing and rendering different pages.
 * @returns {JSX.Element} The JSX element representing the application.
 */
export default function App() {
    return (
        <Routes>
            <Route index element={<StartPage />} />
            {/* Route for the start page */}
            <Route path="/" element={<StartPage />} />
            {/* Route for the templates page */}
            <Route path="templates" element={<Templates />} />
            {/* Route To Create a docunment and Provide export Options */}
            <Route path="export" element={<Tiptap />} />
        </Routes>
    );
}
