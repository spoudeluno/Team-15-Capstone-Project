import { MultiSelect } from '../multiSelect';
import WidthSlider from '../widthSlider';
import { useEffect } from 'react';
import { proposalConfiguration, proposal } from '../../DataModels/ProposalConfiguration.js';
import { initializeColumnsToShow, getColumnElements, getColumnWidths } from '../../Functions/multiSelectHelper';


/**
 * A React component that provides which colums to show and the slide.
 *
 * @param {Object} data - The data object containing information about the colums.
 * @returns {JSX.Element} The rendered CountrySelect component.
 */
export default function ColumnCustomization(data) {

    useEffect(() => {
            initializeColumnsToShow();
            proposal.columnElements = getColumnElements();
            proposal.columnWidths = getColumnWidths();
    }, []);

    return (
        <div className="tabContent columnCust">
            <h3 id='desc'>Please select the columns to display</h3>
            <MultiSelect onChange={data.onChange} columnsSelected={proposalConfiguration.columnsToShow} />
            <WidthSlider />
        </div>
    );
}
