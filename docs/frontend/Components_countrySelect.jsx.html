import { proposalConfiguration } from '../DataModels/ProposalConfiguration';

/**
 * A React component that provides a dropdown selection of countries.
 *
 * @param {Object} onChange - An object containing a callback function to be called when the selected country changes.
 * @param {function(string):void} onChange.onChange - The callback function to be called when the selected country changes.
 *
 * @returns {JSX.Element} The rendered CountrySelect component.
 */

export function CountrySelect(onChange) {
    const countryOptions = [
        { value: 'GB', name: 'England' },
        { value: 'US', name: 'United States' },
        { value: 'IN', name: 'India' },
        { value: 'AU', name: 'Australia' },
        { value: 'DE', name: 'Germany' },
        { value: 'DM', name: 'Denmark' },
        { value: 'FI', name: 'Finland' }
    ];

    const css = `
        details {
            position: relative;
            top: 0;
            left: 0;
            right: 0;
            border: 1px solid var(--border-color);
            border-radius: 5px;
            width: 80%;
            padding: 6px 9px;
            font-weight: 500;
            font-size: 1.125rem;
            background-color: #fff;
            margin: 6px;
            margin-left: auto;
            margin-right: auto;
        }
        details summary::marker {
          display: none;
          font-size: 0;
        }

        details summary::-webkit-details-marker {
          display: none;
          font-size: 0;
        }

        details summary::after {
          content: "\\25BC";
          display: inline-block;
          font-size: 0.6rem;
          height: 1rem;
          line-height: 1rem;
          margin-left: 0.5rem;
          position: relative;
          transition: transform 0.25s;
        }

        summary {
            margin: -6px -9px;
            padding: 6px 9px;
        }

        details[open] summary {
          margin-bottom: 1rem;
          border-bottom: 1px solid var(--border-color);
          border-radius: 0 0 5px 5px;
          padding: 6px;
          margin-top: -6px;
          margin-left: -10.222px;
          margin-right: -10.222px;
        }

        details[open] summary::after {
          top: -0.15rem;
          transform: rotate(180deg);
        }

        .radioCont > details > fieldset legend {
          margin-top: -16px !important;
        }
        .radioCont > details {
          z-index: 6;
        }

        fieldset {
          position: relative;
          border: 0;
          padding: 0;
          font-weight: 400;
          font-size: 1.125rem;
          width: calc(100% + 18px);
          margin: -9px;
          margin-bottom: -2px;
          margin-top: -16px;
          max-height: 250px;
          overflow-y: scroll;
          background-color: white;
        }

        fieldset legend {
          position: absolute !important;
          width: 1px !important;
          height: 1px !important;
          padding: 0 !important;
          margin: -1px !important;
          overflow: hidden !important;
          clip: rect(0, 0, 0, 0) !important;
          white-space: nowrap !important;
          border: 0 !important;
        }

        ul {
          list-style: none;
          margin: 0;
          padding: 0;
        }

        ul li {
          display: flex;
          border-radius: 3px;
          margin: 0;
          padding: 8px;
        }

        ul li:hover {
          background-color: var(--bt-teal-15) !important;
        }

        li:has(input[type=checkbox]:checked) {
          background-color: var(--bt-teal-t10);
        }

        ul li label {
          float: left;
          padding-left: 24px;
          display: grid;
          grid-template-columns: 2fr 1fr;
          width: 100%;
        }

        .pointer {
            cursor: pointer;

            > label
            {
                pointer-events: none;
            }
        }

        input[type='radio'] {
            justify-self: center;
            align-self: center;
            height: 60%;
            width: 60%;
        }

        .radioCont {
            width: 100%;
            height: 0px;
            padding: 12px;
            z-index: 5;
        }
    `;
    function handleCountryChange(value) {
        proposalConfiguration.countryCode = value;
        onChange.onChange();
    }

    function select(value, name) {
        const detail = document.getElementsByTagName('details')[0];
        const summary = document.getElementsByTagName('summary')[0];
        const radio = document.getElementById(value);
        radio.checked = true;
        summary.textContent = name + '  ';
        detail.open = false;
        handleCountryChange(value);
    }

    return (
        <div className="radioCont">
            <style>{css}</style>
            <details className="pointer noHighlight">
                <summary>Change location  </summary>
                <fieldset>
                    <ul>
                        {countryOptions.map(option => (
                            <li key={option.value} onClick={() => select(option.value, option.name)} className="pointer">
                                <label>
                                    <label className="unclickable">{option.name}</label>
                                    <input
                                        id={option.value}
                                        type="radio"
                                        name="country"
                                        value={option.value}
                                        onChange={() => handleCountryChange(option.value)}
                                        checked={option.value === proposalConfiguration.countryCode}
                                    />
                                </label>
                            </li>
                        ))}
                    </ul>
                </fieldset>
            </details>
        </div>
    );
}
